DAX AND MEASURES

 

1. Total Attacks

Total Attacks = CALCULATE( SUM('Datasets Plots'[Non-Prevention Count])) + SUM('Datasets Plots'[Prevention Count])

 

2. Attacks/Month

Attacks/Month = DIVIDE([Total Attacks],117)

 

3. Attacks/Month LY

Attack/Month LY = CALCULATE(

                            [Attacks/Month], DATEADD(Calender[Date],-1,YEAR

                           ))

 

4. Arrow Attacks/Month

Arrow Attacks/Month = SWITCH(TRUE(),

                                [YOY] < 0, UNICHAR(9660), UNICHAR(9650)

)

 

 

 

5. Color Arrow

ColorArrow YOY = IF([Arrow Attacks/Month] = UNICHAR(9660), "Green", "Red")

 

6. YOY

YOY = DIVIDE([Attacks/Month] - [Attack/Month LY], [Attack/Month LY])

 

7. Calender

Calender = CALENDAR(DATE(2001,08,16), DATE(2017,01,11))

 

8. Days of the Week

Days of Week = FORMAT(Calender[Date], "DDDD")

 

9. Month

Month = FORMAT(Calender[Date], "M")

 

10.             Month Name

MonthName = FORMAT(Calender[Date], "MMMM")

11.             Year

Year = (YEAR(Calender[Date]))
